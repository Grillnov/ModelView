141 template<typename GLclientside>
142 void BufferPack<GLclientside>::Attach(GLenum hintBindingPoint, GLenum hintUsage)
143 {
144  if (this->isAttached)
145  {
146  Warning(debugMsg, "Buffer %u is already attached, bailing.", this->AssetID);
147  return;
148  }
149  glGenBuffers(1, &this->AssetID);
150 
151  GLsizei bufferSize = this->num_of_elements * sizeof(GLclientside);
152 
153  glBindBuffer(hintBindingPoint, this->AssetID);
154  glBufferData(hintBindingPoint, bufferSize, nullptr, hintUsage);
155 
156  glBindBuffer(hintBindingPoint, 0);
157 
158  CheckStatus(__FUNCTION__);
159  Log(debugMsg, "Successfully allocated %d bytes of data on server side for buffer %u.", bufferSize, this->AssetID);
160 
161  Log(debugMsg, "Buffer %u was successfully attached.", this->AssetID);
162  this->isAttached = true;
163 }

204 template<typename GLclientside>
205 GLclientside& BufferPack<GLclientside>::operator[](GLsizei index)
206 {
207  if (index > this->num_of_elements)
208  {
209  Error(debugMsg, "Cannot access %dth element in buffer %u, out of boundary!", index, this->AssetID);
210  }
211  if (!this->isAttached)
212  {
213  Error(debugMsg, "Buffer %u is not attached yet!", this->AssetID);
214  }
215  if (this->isMapped)
216  {
217  return *(this->MappedPtr + index);
218  }
219  else
220  {
221  glBindBuffer(GL_COPY_WRITE_BUFFER, this->AssetID);
222  this->MappedPtr = reinterpret_cast<GLclientside*>(glMapBuffer(GL_COPY_WRITE_BUFFER, GL_READ_WRITE));
223 
224  if (MappedPtr == nullptr)
225  Error(debugMsg, "Cannot map buffer %u.", this->AssetID);
226 
227  CheckStatus(__FUNCTION__);
228 
229  this->isMapped = true;
230  return *(this->MappedPtr + index);
231  }
